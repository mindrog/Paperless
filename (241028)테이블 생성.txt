DROP database paperless;
CREATE DATABASE IF NOT EXISTS paperless;
USE paperless;

-- 1. 기존 테이블 삭제
DROP TABLE IF EXISTS `Scheduler`;
DROP TABLE IF EXISTS `Recipient`;
DROP TABLE IF EXISTS `Reference`;
DROP TABLE IF EXISTS `Approver`;
DROP TABLE IF EXISTS `Attachment`;
DROP TABLE IF EXISTS `PurchaseReport`;
DROP TABLE IF EXISTS `AttendanceReport`;
DROP TABLE IF EXISTS `WorkReport`;
DROP TABLE IF EXISTS `Inquiry`;
DROP TABLE IF EXISTS `Stock`;
DROP TABLE IF EXISTS `Emailmessage`;
DROP TABLE IF EXISTS `Employee`;
DROP TABLE IF EXISTS `Position`;
DROP TABLE IF EXISTS `Department`;
DROP TABLE IF EXISTS `Company`;

-- 2. 테이블 생성

-- Company 테이블 생성
CREATE TABLE `Company` (
    `comp_no` BIGINT NOT NULL AUTO_INCREMENT,
    `comp_name` VARCHAR(300) NOT NULL,
    `comp_industry` VARCHAR(100) NOT NULL,
    `comp_requester` VARCHAR(100) NOT NULL,
    `comp_email` VARCHAR(300) NOT NULL,
    `comp_phone` VARCHAR(100) NOT NULL,
    `comp_headcount` INT NOT NULL DEFAULT 0,
    PRIMARY KEY (`comp_no`)
);

-- Department 테이블 생성
CREATE TABLE `Department` (
    `dept_no` BIGINT NOT NULL,
    `dept_name` VARCHAR(100) NOT NULL,
    `dept_team_name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`dept_no`)
);

-- Position 테이블 생성
CREATE TABLE `Position` (
    `posi_no` BIGINT NOT NULL AUTO_INCREMENT,
    `posi_name` VARCHAR(100) NOT NULL,
    PRIMARY KEY (`posi_no`)
);

-- Employee 테이블 생성
CREATE TABLE `Employee` (
    `emp_no` BIGINT NOT NULL AUTO_INCREMENT,
    `emp_code` VARCHAR(100) NOT NULL,
    `emp_pw` VARCHAR(100) NOT NULL,
    `emp_name` VARCHAR(100) NOT NULL,
    `emp_email` VARCHAR(300) NOT NULL,
    `emp_phone` VARCHAR(100) NOT NULL,
    `emp_sign` VARCHAR(100) NULL,
    `emp_profile` VARCHAR(300) NULL,
    `emp_comp_no` BIGINT NOT NULL,
    `emp_dept_no` BIGINT NOT NULL,
    `emp_posi_no` BIGINT NOT NULL,
    `emp_enroll_date` TIMESTAMP NOT NULL DEFAULT NOW(),
    `emp_join_date` TIMESTAMP NOT NULL,
    `emp_role` VARCHAR(100) NOT NULL DEFAULT 'user',
    PRIMARY KEY (`emp_no`),
    FOREIGN KEY (`emp_comp_no`) REFERENCES `Company` (`comp_no`),
    FOREIGN KEY (`emp_dept_no`) REFERENCES `Department` (`dept_no`),
    FOREIGN KEY (`emp_posi_no`) REFERENCES `Position` (`posi_no`)
);

-- Report 테이블 생성
CREATE TABLE `Report` (
    `repo_no` BIGINT NOT NULL AUTO_INCREMENT,
    `repo_emp_no` BIGINT NOT NULL,
    `repo_title` VARCHAR(300) NOT NULL,
    `repo_content` TEXT NOT NULL,
    `repo_status` VARCHAR(100) NOT NULL DEFAULT 'pending',
    `repo_type` VARCHAR(100) NOT NULL,
    `submission_date` TIMESTAMP NULL,
    `cancel_date` TIMESTAMP NULL,
    PRIMARY KEY (`repo_no`),
    FOREIGN KEY (`repo_emp_no`) REFERENCES `Employee` (`emp_no`)
);

-- WorkReport 테이블 생성
CREATE TABLE `WorkReport` (
    `work_repo_no` BIGINT NOT NULL,
    `repo_start_time` TIMESTAMP NULL,
    `repo_end_time` TIMESTAMP NULL,
    PRIMARY KEY (`work_repo_no`),
    FOREIGN KEY (`work_repo_no`) REFERENCES `Report` (`repo_no`)
);

-- AttendanceReport 테이블 생성
CREATE TABLE `AttendanceReport` (
    `atten_repo_no` BIGINT NOT NULL,
    `vacation_type` VARCHAR(50) NOT NULL,
    `atten_start_date` DATE NOT NULL,
    `atten_end_date` DATE NULL,
    PRIMARY KEY (`atten_repo_no`),
    FOREIGN KEY (`atten_repo_no`) REFERENCES `Report` (`repo_no`)
);

-- PurchaseReport 테이블 생성
CREATE TABLE `PurchaseReport` (
    `purch_repo_no` BIGINT NOT NULL,
    `product_name` VARCHAR(255) NOT NULL,
    PRIMARY KEY (`purch_repo_no`),
    FOREIGN KEY (`purch_repo_no`) REFERENCES `Report` (`repo_no`)
);

-- Approver 테이블 생성
CREATE TABLE `Approver` (
    `appr_no` BIGINT NOT NULL AUTO_INCREMENT,
    `appr_repo_no` BIGINT NOT NULL,
    `appr_emp_no` BIGINT NOT NULL,
    `appr_status` VARCHAR(100) NOT NULL DEFAULT 'pending',
    PRIMARY KEY (`appr_no`),
    FOREIGN KEY (`appr_repo_no`) REFERENCES `Report` (`repo_no`),
    FOREIGN KEY (`appr_emp_no`) REFERENCES `Employee` (`emp_no`)
);

-- Recipient 테이블 생성
CREATE TABLE `Recipient` (
    `reci_no` BIGINT NOT NULL,
    `reci_repo_no` BIGINT NOT NULL,
    `reci_emp_no` BIGINT NOT NULL,
    PRIMARY KEY (`reci_no`),
    FOREIGN KEY (`reci_repo_no`) REFERENCES `Report` (`repo_no`),
    FOREIGN KEY (`reci_emp_no`) REFERENCES `Employee` (`emp_no`)
);

-- Reference 테이블 생성
CREATE TABLE `Reference` (
    `refe_no` BIGINT NOT NULL,
    `refe_repo_no` BIGINT NOT NULL,
    `refe_emp_no` BIGINT NOT NULL,
    PRIMARY KEY (`refe_no`),
    FOREIGN KEY (`refe_repo_no`) REFERENCES `Report` (`repo_no`),
    FOREIGN KEY (`refe_emp_no`) REFERENCES `Employee` (`emp_no`)
);

-- Attachment 테이블 (첨부파일 기본 정보 저장)
CREATE TABLE `Attachment` (
    `atta_no` BIGINT NOT NULL AUTO_INCREMENT,
    `atta_key` VARCHAR(500) NOT NULL,
    `atta_url` VARCHAR(1000) NOT NULL,
    `atta_original_name` VARCHAR(500) NOT NULL,
    `atta_size` BIGINT NOT NULL,
    PRIMARY KEY (`atta_no`)
);

-- Emailmessage 테이블 생성
CREATE TABLE `Emailmessage` (
    `email_no` BIGINT NOT NULL AUTO_INCREMENT,
    `email_writer` BIGINT NOT NULL,
    `email_title`  varchar(100) not null,
    `email_content` VARCHAR(500) NOT NULL,
    `email_recipient` BIGINT NOT NULL,
    `email_cc` BIGINT DEFAULT NULL,
    `email_status` VARCHAR(100) NOT NULL DEFAULT 'unread',
    `email_send_date` TIMESTAMP NOT NULL DEFAULT NOW(),
    PRIMARY KEY (`email_no`),
    FOREIGN KEY (`email_writer`) REFERENCES `Employee` (`emp_no`),
    FOREIGN KEY (`email_recipient`) REFERENCES `Employee` (`emp_no`),
    FOREIGN KEY (`email_cc`) REFERENCES `Employee` (`emp_no`)
);

-- EmailAttachment 테이블 (이메일과 첨부파일의 관계)
CREATE TABLE `EmailAttachment` (
    `email_no` BIGINT NOT NULL,
    `atta_no` BIGINT NOT NULL,
    PRIMARY KEY (`email_no`, `atta_no`),
    FOREIGN KEY (`email_no`) REFERENCES `Emailmessage` (`email_no`),
    FOREIGN KEY (`atta_no`) REFERENCES `Attachment` (`atta_no`)
);

-- ChatAttachment 테이블 (채팅 첨부파일)
CREATE TABLE `ChatAttachment` (
    `chat_atta_no` BIGINT NOT NULL,
    `chat_atta_key` VARCHAR(500) NOT NULL,
    `chat_atta_url` TEXT NOT NULL,
    `chat_atta_name` VARCHAR(500) NOT NULL,
    PRIMARY KEY (`chat_atta_no`)
);

-- Scheduler 테이블 생성
CREATE TABLE `Scheduler` (
    `sche_no` BIGINT NOT NULL,
    `sche_emp_no` BIGINT NOT NULL,
    PRIMARY KEY (`sche_no`),
    FOREIGN KEY (`sche_emp_no`) REFERENCES `Employee` (`emp_no`)
);

-- Stock 테이블 생성
CREATE TABLE `Stock` (
    `stock_no` BIGINT NOT NULL,
    `stock_comp_no` BIGINT NOT NULL,
    `stock_name` VARCHAR(500) NOT NULL,
    `stock_category` VARCHAR(500) NOT NULL,
    `stock_quantity`BIGINT NOT NULL,
    `stock_price` BIGINT NOT NULL,
    PRIMARY KEY (`stock_no`),
    FOREIGN KEY (`stock_comp_no`) REFERENCES `Company` (`comp_no`)
);

-- Inquiry 테이블 생성
CREATE TABLE `Inquiry` (
    `inqu_no` BIGINT NOT NULL AUTO_INCREMENT,
    `inqu_writer` VARCHAR(100) NOT NULL,
    `inqu_compName` VARCHAR(100) NOT NULL,
    `inqu_compType` VARCHAR(100) NOT NULL,
    `inqu_email` VARCHAR(100) NOT NULL,
    `inqu_phone` VARCHAR(100) NOT NULL,
    `inqu_numberOfPeople` VARCHAR(100) NOT NULL,
    `inqu_enroll` timestamp NOT NULL,
    
    PRIMARY KEY (`inqu_no`)
);

-- `Request`  테이블 생성
CREATE TABLE `Request` (
    `requ_no` BIGINT NOT NULL AUTO_INCREMENT,
    `requ_title` VARCHAR(100) NOT NULL,
    `requ_writer` VARCHAR(100) NOT NULL,
    `requ_phone` VARCHAR(100) NOT NULL,
    `requ_email` VARCHAR(100) NOT NULL,
    `requ_contents` TEXT NOT NULL,
    `requ_enroll` timestamp NOT NULL,
    
    PRIMARY KEY (`requ_no`)
);
